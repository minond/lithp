number    : /-?[0-9]+/ ;

symbol    : '+' | '-' | '*' | '/' |
            "list" | "head" | "tail" | "join" | "eval" |
            "len" | "cons"
            ;

sexpr     : '(' <expr>* ')' ;
qexpr     : '{' <expr>* '}' ;
expr      : <number> | <symbol> | <sexpr> | <qexpr> ;

lithp     : /^/ <expr>* /$/ ;
